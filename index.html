    <script>

        function toggleDay(day) {

            document.querySelectorAll('.tab').forEach(tab => tab.classList.remove('active'));

            document.getElementById(day).classList.add('active');

        }


        document.querySelectorAll('.reps').forEach(repsDiv => {

            repsDiv.querySelectorAll('button').forEach(btn => {

                btn.onclick = () => {

                    repsDiv.querySelectorAll('button').forEach(b => b.classList.remove('selected'));

                    btn.classList.add('selected');

                };

            });

        });


        function saveWorkout(day) {

            const form = document.getElementById(day + '-form');

            const data = { day, date: new Date().toISOString(), exercises: {} };

            const inputs = form.querySelectorAll('input[type="number"]');

            inputs.forEach(input => {

                if (input.value) data.exercises[input.name] = input.value;

            });

            document.querySelectorAll('.reps').forEach(repsDiv => {

                const selected = repsDiv.querySelector('.selected');

                if (selected) data.exercises[repsDiv.dataset.set] = selected.textContent;

            });

            let workouts = JSON.parse(localStorage.getItem('workouts') || '[]');

            workouts.push(data);

            localStorage.setItem('workouts', JSON.stringify(workouts));

            alert('Workout saved!');

            form.reset();

            document.querySelectorAll('.selected').forEach(s => s.classList.remove('selected'));

        }


        function showHistory() {

            document.querySelectorAll('.tab').forEach(tab => tab.classList.remove('active'));

            document.getElementById('history').classList.add('active');

            const workouts = JSON.parse(localStorage.getItem('workouts') || '[]');

            const list = document.getElementById('history-list');

            list.innerHTML = '';

            workouts.forEach((workout, index) => {

                const li = document.createElement('li');

                li.innerHTML = `

                    <div>${workout.day.toUpperCase()} - ${new Date(workout.date).toLocaleString()}</div>

                    <button onclick="toggleDetails(${index})">View Details</button>

                    <button onclick="deleteWorkout(${index})">Delete</button>

                    <div id="details-${index}" class="history-details"></div>

                `;

                list.appendChild(li);

            });

        }


        function toggleDetails(index) {

            const details = document.getElementById(`details-${index}`);

            const workout = JSON.parse(localStorage.getItem('workouts'))[index];

            if (details.style.display === 'block') {

                details.style.display = 'none';

            } else {

                let html = '<ul>';

                for (let key in workout.exercises) {

                    html += `<li>${key}: ${workout.exercises[key]}</li>`;

                }

                html += '</ul>';

                details.innerHTML = html;

                details.style.display = 'block';

            }

        }


        function deleteWorkout(index) {

            let workouts = JSON.parse(localStorage.getItem('workouts'));

            workouts.splice(index, 1);

            localStorage.setItem('workouts', JSON.stringify(workouts));

            showHistory();

        }


        function hideHistory() {

            document.querySelectorAll('.tab').forEach(tab => tab.classList.remove('active'));

            document.getElementById('day1').classList.add('active');

        }
